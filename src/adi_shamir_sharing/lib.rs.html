1//! # Shamir's Secret Sharing Implementation in Rust
2//!
3//! This crate provides an implementation of Shamir's Secret Sharing algorithm over a finite field.
4//! It splits a secret into multiple shares using a random polynomial and can reconstruct the secret
5//! from a subset of those shares using Lagrange interpolation.
6//!
7//! The implementation uses the prime modulus defined by the secp256k1 curve:
8//! \( p = 2^{256} - 2^{32} - 977 \).
9//!
10//! ## How It Works
11//!
12//! 1. **Field Selection:**  
13//!    A prime \( p \) is chosen to define the finite field over which all computations occur.
14//!
15//! 2. **Polynomial Construction:**  
16//!    A random polynomial \( f(x) = a_0 + a_1 x + a_2 x^2 + \dots + a_{k-1} x^{k-1} \) is generated,
17//!    where \( a_0 \) is the secret and the other coefficients \( a_i \) are generated randomly.
18//!
19//! 3. **Share Generation:**  
20//!    The polynomial is evaluated at \( n \) distinct nonzero x-values to yield shares \( (x_i, f(x_i)) \).
21//!
22//! 4. **Secret Reconstruction:**  
23//!    Given at least \( k \) shares, Lagrange interpolation is used to reconstruct \( f(0) \), which is the secret.
24//!
25//! ## Security
26//!
27//! Shamir's Secret Sharing is information-theoretically secure. Any set of fewer than \( k \) shares
28//! yields no information about the secret due to the randomness of the polynomial coefficients.
29//!
30//! **Important considerations:**
31//! - **Randomness:** The security relies on using a cryptographically secure random number generator for
32//!   coefficient generation.
33//! - **Parameter Selection:** The secret must be less than \( p \), and the threshold \( k \) should be chosen
34//!   according to the desired security policy.
35//!
36//! ## Potential Vulnerabilities
37//!
38//! - **Weak Randomness:** If the random number generator is compromised or not cryptographically secure,
39//!   an attacker may recover the random coefficients and, therefore, the secret.
40//!
41//! - **Side Channel Attacks:** As with many cryptographic algorithms, careful attention must be paid to avoid
42//!   side channel leaks (timing, power, etc.), especially in a constrained or adversarial environment.
43//!
44//! - **Misconfiguration:** Incorrect parameters (e.g., secret larger than \( p \) or an invalid threshold)
45//!   can cause the scheme to fail or reduce security.
46
47use num_bigint::{BigUint, ToBigUint, RandBigInt};
48use num_traits::{One, Zero, Num};
49use rand::thread_rng;

51/// Represents a single share in Shamir's Secret Sharing.
52///
53/// Each share is a point \((x, y)\) on the polynomial.
54#[derive(Debug, Clone)]
55pub struct Share {
56    /// The x-coordinate of the share.
57    pub x: BigUint,
58    /// The y-coordinate of the share.
59    pub y: BigUint,
60}
61
62/// Returns the prime modulus \( p = 2^{256} - 2^{32} - 977 \).
63///
64/// This prime is used in secp256k1 and is large enough for 256-bit secrets, ensuring that
65/// all arithmetic operations remain valid in the finite field.
66///
67/// # Returns
68/// A `BigUint` representing the prime modulus.
69pub fn get_prime() -> BigUint {
70    BigUint::from_str_radix(
71        "fffffffffffffffffffffffffffffffffffffffffffffffffffffffefffffc2f",
72        16,
73    )
74    .unwrap()
75}

77/// Computes the modular inverse of `a` modulo `p` using Fermat's Little Theorem.
78///
79/// Given that `p` is prime, the modular inverse is computed as:
80/// \[ a^{-1} \equiv a^{p-2} \mod p \]
81///
82/// # Arguments
83///
84/// * `a` - The number for which the inverse is computed.
85/// * `p` - The prime modulus.
86///
87/// # Returns
88/// The modular inverse of `a` modulo `p`.
89pub fn modinv(a: &BigUint, p: &BigUint) -> BigUint {
90    a.modpow(&(p - 2u32.to_biguint().unwrap()), p)
91}
92
93/// Evaluates a polynomial at a given point \( x \) modulo \( p \).
94///
95/// The polynomial is given by its coefficients in ascending order:
96/// \[ f(x) = \sum_{i=0}^{n-1} \text{coeffs}[i] \cdot x^i \mod p \]
97///
98/// # Arguments
99///
100/// * `coeffs` - A slice of `BigUint` containing the polynomial coefficients.
101/// * `x` - The evaluation point.
102/// * `p` - The prime modulus.
103///
104/// # Returns
105/// The result of the polynomial evaluation \( f(x) \mod p \).
106pub fn evaluate_polynomial(coeffs: &[BigUint], x: &BigUint, p: &BigUint) -> BigUint {
107    let mut result = BigUint::zero();
108    let mut power_of_x = BigUint::one();
109    for coeff in coeffs {
110        result = (result + (coeff * &power_of_x) % p) % p;
111        power_of_x = (power_of_x * x) % p;
112    }
113    result
114}

116/// Splits a secret into a specified number of shares with a reconstruction threshold.
117///
118/// A random polynomial of degree `threshold - 1` is constructed with the secret as the constant term.
119/// The polynomial is then evaluated at different nonzero x-values to produce the shares.
120///
121/// # Arguments
122///
123/// * `secret` - The secret to be shared. Must be less than `p`.
124/// * `threshold` - The minimum number of shares required to reconstruct the secret.
125/// * `share_count` - The total number of shares to generate.
126/// * `p` - The prime modulus.
127///
128/// # Returns
129/// * `Ok(Vec<Share>)` containing the generated shares, or an error string if parameters are invalid.
130pub fn split_secret(
131    secret: &BigUint,
132    threshold: usize,
133    share_count: usize,
134    p: &BigUint,
135) -> Result<Vec<Share>, String> {
136    if threshold > share_count {
137        return Err("Threshold cannot be greater than the number of shares".to_string());
138    }
139    if secret >= p {
140        return Err("Secret must be less than the prime modulus".to_string());
141    }
142    let mut rng = thread_rng();
143    let mut coeffs: Vec<BigUint> = vec![secret.clone()];
144    for _ in 1..threshold {
145        coeffs.push(rng.gen_biguint_below(p));
146    }
147    let mut shares: Vec<Share> = Vec::new();
148    for i in 1..=share_count {
149        let x = BigUint::from(i);
150        let y = evaluate_polynomial(&coeffs, &x, p);
151        shares.push(Share { x, y });
152    }
153    Ok(shares)
154}

156/// Reconstructs the secret from a given set of shares using Lagrange interpolation.
157///
158/// Lagrange interpolation is used to compute \( f(0) \), which recovers the secret.
159/// Each share contributes a Lagrange basis polynomial, and the sum of the contributions
160/// gives the original secret.
161///
162/// # Arguments
163///
164/// * `shares` - A slice containing at least `threshold` shares.
165/// * `p` - The prime modulus.
166///
167/// # Returns
168/// The reconstructed secret \( f(0) \).
169pub fn reconstruct_secret(shares: &[Share], p: &BigUint) -> BigUint {
170    let mut secret = BigUint::zero();
171    let k = shares.len();
172    for j in 0..k {
173        let mut numerator = BigUint::one();
174        let mut denominator = BigUint::one();
175        for m in 0..k {
176            if m != j {
177                let xm = &shares[m].x;
178                numerator = (&numerator * (p - xm)) % p;
179                let xj = &shares[j].x;
180                let diff = if xj >= xm { xj - xm } else { p - (xm - xj) };
181                denominator = (denominator * diff) % p;
182            }
183        }
184        let lambda = (&numerator * modinv(&denominator, p)) % p;
185        secret = (secret + (&shares[j].y * lambda)) % p;
186    }
187    secret
188}
